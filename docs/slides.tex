%==============================================================================
\section{Introducci\'on}
%==============================================================================

\begin{frame}{¿Qu\'e es el ray tracing?}
\begin{columns}[T]
\begin{column}{0.58\textwidth}
\begin{itemize}\itemsep6pt
  \item T\'ecnica de renderizado que \hl{sigue el camino de la luz}: por cada
        p\'ixel se lanza un \emph{rayo} desde la c\'amara hacia la escena.
  \item Al chocar con una superficie se calcula sombra, y se pueden generar
        \hl{rayos secundarios} (reflejo, iluminaci\'on indirecta).
  \item Produce sombras e iluminaci\'on globales de forma f\'isica, a cambio de
        un costo de c\'omputo alto.
\end{itemize}
\end{column}
\begin{column}{0.42\textwidth}\centering
\begin{tikzpicture}[scale=0.85,every node/.style={font=\scriptsize,text=ink}]
  \draw[->,thick,primlt] (-0.2,0) -- (4.8,0.7) node[midway,above]{rayo};
  \node[draw=primlt,fill=panellt,minimum width=1.3cm,minimum height=0.9cm] (a) at (4.8,0.7){};
  \draw[->,thick,prim] (4.8,0.7) -- (3.0,2.4) node[midway,right]{reflejo};
  \node[draw=prim,fill=panellt,circle,minimum size=0.5cm] at (2.8,2.6){};
  \node[mute] at (4.8,-0.3) {superficie};
  \node[mute] at (0,-0.3) {c\'amara};
\end{tikzpicture}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Contexto: d\'onde se usa}
\begin{columns}[T]
\begin{column}{0.6\textwidth}
\begin{itemize}\itemsep4pt
  \item \hl{Cine y animaci\'on:} render de producci\'on (Pixar, ILM).
  \item \hl{Videojuegos:} iluminaci\'on por trazado en tiempo real
        (NVIDIA RTX, consolas actuales).
  \item \hl{Dise\~no y arquitectura:} previsualizaci\'on fotorrealista.
\end{itemize}
\end{column}
\begin{column}{0.4\textwidth}
\begin{alertblock}{El cuello de botella}
El costo domina por el n\'umero de \emph{intersecciones rayo\,--\,geometr\'ia}.
Su viabilidad depende de una \hl{estructura de aceleraci\'on}.
\end{alertblock}
\end{column}
\end{columns}
\vspace{8pt}
\centering\footnotesize\textcolor{mute}{Ese es el punto de contacto con el curso: la estructura de datos \emph{es} lo que lo vuelve viable.}
\end{frame}

%==============================================================================
\section{El problema}
%==============================================================================

\begin{frame}{El ray tracing ingenuo es intratable}
Sin aceleraci\'on, cada rayo se prueba contra \hl{todos} los tri\'angulos:
\[
T_{\text{frame}} \;\approx\; \underbrace{P}_{\text{p\'ixeles}}\cdot
\underbrace{N}_{\text{tri\'angulos}}\cdot(\text{rayos por p\'ixel})
\]
\begin{alertblock}{Orden de magnitud}
$1280\times720$ con $200\,000$ tri\'angulos $\Rightarrow \sim\!1{,}8\times10^{11}$
pruebas por frame. Inviable en tiempo real con fuerza bruta $O(P\cdot N)$.
\end{alertblock}
\begin{itemize}
  \item Necesitamos \hl{descartar grupos enteros} de tri\'angulos con pruebas baratas.
  \item $\Rightarrow$ una estructura de aceleraci\'on jer\'arquica: el \alert{BVH}.
\end{itemize}
\end{frame}

\begin{frame}{Escenas est\'aticas, din\'amicas y mixtas}
\begin{columns}[T]
\begin{column}{0.5\textwidth}
\begin{block}{Geometr\'ia est\'atica}
Edificios, terreno, objetos fijos. El \'arbol se \hl{construye una vez} y se
reutiliza en todos los frames $\Rightarrow$ el build se \emph{amortiza}.
\end{block}
\end{column}
\begin{column}{0.5\textwidth}
\begin{exampleblock}{Geometr\'ia din\'amica}
Personajes, peatones, \emph{mesh} deformado por animaci\'on. Cambia cada frame
$\Rightarrow$ el \'arbol se \hl{reconstruye cada frame} y el build \emph{no} se amortiza.
\end{exampleblock}
\end{column}
\end{columns}
\vspace{5pt}
\begin{alertblock}{La pregunta que gu\'ia el proyecto}
Una escena real es \hl{mixta}: parte fija + parte en movimiento. ¿Qu\'e heur\'istica
de construcci\'on conviene para cada una, y qu\'e \emph{hardware} recorre mejor el \'arbol?
\end{alertblock}
\end{frame}

%==============================================================================
\section{Algoritmo}
%==============================================================================

\begin{frame}{BVH: la idea}
\begin{columns}[T]
\begin{column}{0.54\textwidth}
\begin{itemize}\itemsep3pt
  \item \'Arbol binario de \hl{cajas alineadas a los ejes} (AABB).
  \item La ra\'iz envuelve toda la escena; cada nodo \hl{reparte} sus tri\'angulos en 2 hijos.
  \item Las \hl{hojas} guardan pocos tri\'angulos.
  \item El rayo prueba primero las cajas y solo desciende por las que golpea.
\end{itemize}
\vspace{3pt}
\centering Costo por rayo: $O(N)\;\to\;\sim$\hl{$O(\log N)$}.
\end{column}
\begin{column}{0.46\textwidth}\centering
\begin{tikzpicture}[scale=0.72,every node/.style={font=\scriptsize,text=ink},
  bx/.style={draw=prim,thick,rounded corners=1pt,fill=panel},

  \node[bx,minimum width=1.6cm,minimum height=0.4cm] (b) at (1.8,1.7) {hijo der.};

\end{tikzpicture}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Dos pruebas b\'asicas: AABB y tri\'angulo}
\begin{columns}[T]
\begin{column}{0.5\textwidth}
\textbf{Rayo\,--\,AABB (\emph{slabs}).} Una caja $=$ 3 bandas (X, Y, Z). El rayo
entra sii el intervalo de entrada no supera al de salida:
\[
t_{\min}=\max(\text{ent.})\le t_{\max}=\min(\text{sal.})
\]
Se precalcula $\mathrm{inv}=1/\vecm{D}$ por rayo (la divisi\'on es cara) y luego
solo se multiplica.
\end{column}
\begin{column}{0.5\textwidth}
\textbf{Rayo\,--\,tri\'angulo (M\"oller\textendash Trumbore).} Resuelve
$(t,u,v)$ de $\vecm{O}+t\vecm{D}=\vecm{v_0}+u\,\vecm{e_1}+v\,\vecm{e_2}$.
\begin{itemize}\itemsep1pt
  \item $u,v\ge0,\;u{+}v\le1$ \emph{es} la prueba de estar dentro.
  \item $(u,v)$ son los \hl{pesos baric\'entricos}: interpolan normal, color y UV.
\end{itemize}
\end{column}
\end{columns}
\vspace{3pt}
\centering\footnotesize\textcolor{mute}{Ambas son $O(1)$: el BVH sirve para \emph{invocarlas pocas veces}, no para abaratarlas.}
\end{frame}

\begin{frame}{Construcci\'on \emph{top-down} y las tres heur\'isticas}
\begin{columns}[T]
\begin{column}{0.44\textwidth}
\small Con todos los tri\'angulos en la ra\'iz, mientras un nodo tenga
$>$ hoja m\'inima:
\begin{enumerate}\itemsep1pt
  \item eje de mayor extensi\'on de centroides;
  \item \hl{plano de corte} $\Rightarrow$ \emph{depende de la heur\'istica};
  \item reparto en 2 hijos y recursi\'on.
\end{enumerate}
\vspace{3pt}
\centering\footnotesize\textcolor{mute}{Comparten todo; solo cambia \alert{d\'onde cortar}.}
\end{column}
\begin{column}{0.56\textwidth}
\begin{block}{SAH\hfill\textcolor{white}{$\bigstar$ mejor \'arbol}}
\footnotesize Minimiza costo $=$ \'area$\times$n.$^\circ$tri (\emph{binning}, 12 cubos). Build lento.
\end{block}
\begin{exampleblock}{Median}
\footnotesize Corta por la mediana de centroides. Simple; ignora el tama\~no.
\end{exampleblock}
\begin{block}{Morton}
\footnotesize Ordena por \hl{c\'odigo Z} y parte el \'indice a la mitad. Build rapid\'isimo; \'arbol mediocre.
\end{block}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{SAH: heur\'istica de \'area de superficie}
\textbf{Intuici\'on:} un rayo entra en un hijo con probabilidad
$\propto$ \'area(hijo)/\'area(padre). El costo esperado de un corte es:
\[
C_{\text{split}} = 1 + \frac{A_L\,n_L + A_R\,n_R}{A_{\text{padre}}}
\]
\begin{columns}[T]
\begin{column}{0.52\textwidth}
\textbf{Binning}: en vez de probar los $O(N)$ cortes posibles, se agrupan los
centroides en \alert{12 cubos} y se eval\'uan 11 cortes $\Rightarrow$ build
$O(N\log N)$, cercano al \'optimo.
\end{column}
\begin{column}{0.48\textwidth}\centering
\begin{tikzpicture}[scale=0.62,every node/.style={font=\scriptsize,text=ink}]
  \foreach \i in {0,...,11}{\draw[rule,fill=panel] (\i*0.5,0) rectangle (\i*0.5+0.5,0.7);}
  \foreach \i/\h in {1/0.5,2/0.6,3/0.35,7/0.2,8/0.55,9/0.65,10/0.45}{
    \fill[prim] (\i*0.5,0) rectangle (\i*0.5+0.5,\h);}
  \draw[thick,primdk,dashed] (3.5,-0.1)--(3.5,0.95);
  \node[primdk] at (3.5,1.15) {\scriptsize corte};
\end{tikzpicture}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Median split: partir por la mitad de los centroides}
\begin{columns}[T]
\begin{column}{0.56\textwidth}
\textbf{Intuici\'on:} dejar \hl{la misma cantidad} de tri\'angulos a cada lado.
\begin{enumerate}\itemsep2pt
  \item Eje de mayor extensi\'on de centroides.
  \item \hl{Quickselect} reordena los tri\'angulos: la primera mitad son los de
        centroide menor en ese eje.
  \item Corte en el \hl{\'indice medio} (mitad/mitad).
\end{enumerate}
\vspace{2pt}
\footnotesize\textcolor{mute}{Build $O(N)$ por nivel. Ignora el \emph{tama\~no} de
las cajas y el espacio vac\'io $\Rightarrow$ hijos que se solapan $\Rightarrow$
traversal peor que SAH.}
\end{column}
\begin{column}{0.44\textwidth}\centering
\begin{tikzpicture}[scale=0.9,every node/.style={font=\scriptsize,text=ink}]
  \foreach \x in {0.2,0.5,0.9,1.5,2.0,2.3,2.9,3.4}{\fill[prim] (\x,0) circle (2pt);}
  \draw[thick,primdk,dashed] (2.15,-0.3)--(2.15,0.6);
  \node[primdk] at (2.15,0.85) {mediana};
  \node[mute] at (1.0,-0.55) {4 izq.};
  \node[mute] at (2.9,-0.55) {4 der.};
  \draw[->,mute] (-0.1,-0.15)--(3.6,-0.15) node[right]{eje largo};
\end{tikzpicture}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Morton / LBVH: ordenar por una curva que llena el espacio}
\begin{columns}[T]
\begin{column}{0.54\textwidth}
\textbf{Intuici\'on:} recorrer el espacio 3D con la \hl{curva Z}, de modo que
tri\'angulos \emph{vecinos} queden \emph{contiguos} en un arreglo; luego partir por
la mitad del \'indice.
\begin{enumerate}\itemsep1pt
  \item Centroide $\to$ normalizar a $[0,1]^3$; cuantizar cada eje a \hl{10 bits}.
  \item \hl{Intercalar} los bits de $x,y,z$: c\'odigo Morton de 30 bits
        ($\dots x_1y_1z_1x_0y_0z_0$).
  \item \hl{Ordenar} una vez por ese c\'odigo y cortar por el \'indice medio.
\end{enumerate}
\vspace{2pt}
\footnotesize\textcolor{mute}{Build rapid\'isimo (domina el \emph{sort}); los cortes
por \'indice ignoran \'area y solapamiento $\Rightarrow$ \'arbol mediocre. Es la base
de los constructores de BVH en GPU.}
\end{column}
\begin{column}{0.46\textwidth}\centering
\begin{tikzpicture}[scale=0.62,every node/.style={font=\tiny,text=ink}]
  \foreach \x in {0,1,2,3}\foreach \y in {0,1,2,3}{\fill[rule] (\x*0.7,\y*0.7) circle (1.4pt);}
  \draw[very thick,prim]
    (0,0)--(0.7,0)--(0,0.7)--(0.7,0.7)--(1.4,0)--(2.1,0)--(1.4,0.7)--(2.1,0.7)--
    (0,1.4)--(0.7,1.4)--(0,2.1)--(0.7,2.1)--(1.4,1.4)--(2.1,1.4)--(1.4,2.1)--(2.1,2.1);
  \node[primdk] at (1.05,-0.5) {curva Z (orden Morton)};
\end{tikzpicture}
\end{column}
\end{columns}
\end{frame}

\begin{frame}[fragile]{Recorrido con pila expl\'icita y poda}
\begin{columns}[T]
\begin{column}{0.5\textwidth}
\begin{itemize}\itemsep2pt
  \item Recorrido \hl{iterativo} con pila fija (sin recursi\'on).
  \item \hl{Poda 1:} caja no golpeada $\Rightarrow$ se salta el sub\'arbol.
  \item \hl{Poda 2:} caja m\'as lejos que el mejor $t$ $\Rightarrow$ se salta.
  \item En la hoja: M\"oller\textendash Trumbore a sus pocos tri\'angulos.
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\begin{lstlisting}
stack.push(raiz);
while (pila no vacia) {
  nodo = pop();
  if (!aabb_hit(nodo.box, inv, best))
      continue;               // poda
  if (es_hoja) test_tris(nodo);
  else         push(hijos);
}
\end{lstlisting}
\end{column}
\end{columns}
\vspace{2pt}
\centering\small La misma rutina, con una pila expl\'icita, es la que luego portamos a la GPU.
\end{frame}

\begin{frame}{Nuestra combinaci\'on: dos BVH (est\'atico $+$ din\'amico)}
\begin{columns}[T]
\begin{column}{0.5\textwidth}
\begin{block}{BVH est\'atico \textemdash{} \textcolor{white}{SAH}}
\begin{itemize}\itemsep1pt
  \item Escena fija (campo, edificios).
  \item Se construye \hl{una vez} con SAH (mejor \'arbol; el build se amortiza).
\end{itemize}
\end{block}
\end{column}
\begin{column}{0.5\textwidth}
\begin{exampleblock}{BVH din\'amico \textemdash{} Morton}
\begin{itemize}\itemsep1pt
  \item Objetos que se mueven / \emph{mesh} skinneado.
  \item Se \hl{reconstruye cada frame}; Morton por su build barato.
\end{itemize}
\end{exampleblock}
\end{column}
\end{columns}
\vspace{5pt}
\centering\small Cada rayo consulta \hl{ambos} \'arboles y se queda con el golpe m\'as cercano.\\[2pt]
\footnotesize\textcolor{mute}{Es una TLAS de dos instancias hecha a mano: la heur\'istica se elige seg\'un la frecuencia de cambio.}
\end{frame}

\section{Entorno de implementaci\'on}
%==============================================================================

\begin{frame}{Tres \emph{backends} sobre la misma escena}
\centering
\begin{tikzpicture}[
  node distance=4mm and 8mm,
  box/.style={draw=prim,thick,rounded corners=2pt,fill=panel,
              minimum width=3.0cm,minimum height=0.9cm,align=center,font=\scriptsize,text=ink},

  \node[box,below left=of scene]  (cpu) {\textbf{CPU propio}\\multihilo, C++};

\vspace{4pt}
\begin{itemize}\itemsep2pt
  \item El mismo raytracer (M\"oller\textendash Trumbore, sombreado, NEE) corre en los tres.
  \item CPU y GPU recorren \hl{nuestro} \'arbol; Embree construye el suyo.
  \item Se cambia de backend \hl{en vivo} con la tecla \texttt{G}.
\end{itemize}
\end{frame}

\begin{frame}{¿Qu\'e es cada \emph{backend}?}
\begin{table}\centering\small
\renewcommand{\arraystretch}{1.5}
\setlength{\tabcolsep}{6pt}
\begin{tabular}{@{}l l l l@{}}
\toprule
\textbf{Backend} & \textbf{Qui\'en lo escribi\'o} & \textbf{D\'onde corre} & \textbf{Qu\'e \'arbol usa} \\
\midrule
\hl{CPU propio} & Nosotros (C++)        & CPU, 16 hilos      & \textbf{nuestro} BVH \\
\hl{GPU OpenCL} & Nosotros (\emph{kernel}) & GPU, 1 hilo/p\'ixel & \textbf{nuestro} (aplanado) \\
\rowcolor{panellt}
\textbf{Embree} & Intel (librer\'ia)    & CPU, SIMD (AVX)    & \textbf{su propio} BVH \\
\bottomrule
\end{tabular}
\end{table}
\vspace{2pt}
\begin{itemize}\itemsep1pt
  \item CPU y GPU son \hl{el mismo raytracer nuestro}: cambia solo el hardware que recorre \emph{nuestro} \'arbol.
  \item Embree es la \hl{referencia industrial} (no es c\'odigo nuestro): arma y recorre su propio \'arbol.
  \item Los tres son \hl{ray tracing por software}: ninguno usa n\'ucleos RT dedicados.
\end{itemize}
\end{frame}

\begin{frame}[fragile]{GPU: por qu\'e el algoritmo cambia (\emph{flatten})}
\begin{columns}[T]
\begin{column}{0.5\textwidth}
\small La GPU \hl{no es un CPU r\'apido}: sin punteros de host ni recursi\'on.
Hay que \hl{aplanar} el \'arbol a arrays paralelos:
\begin{itemize}\itemsep3pt
  \item \texttt{node\_bounds}: 2 \texttt{float4}/nodo (min, max).
  \item \texttt{node\_links}: \texttt{int4} $=$ (izq, der, inicio, cuenta); \texttt{izq<0} $\Rightarrow$ hoja.
  \item \texttt{tris}: 32 \texttt{float}/tri\'angulo.
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\begin{lstlisting}
int stack[64]; int sp=0;
stack[sp++]=0;              // raiz
while(sp>0){
  int4 lk = node_links[stack[--sp]];
  if(lk.x<0) test_tris(lk.z,lk.w);
  else { push(lk.x); push(lk.y); }
}
\end{lstlisting}
\end{column}
\end{columns}
\vspace{4pt}
El kernel recorre con una \hl{pila expl\'icita}; modelo \hl{SIMT}: un
\emph{work-item} por p\'ixel (grupos $8\times8$, rayos vecinos comparten cach\'e).
\end{frame}

\begin{frame}{Embree: qu\'e hace con el BVH}
\begin{columns}[T]
\begin{column}{0.54\textwidth}
\begin{itemize}\itemsep3pt
  \item \hl{MBVH}: nodos de \hl{4\,--\,8 v\'ias} (no binario) $\Rightarrow$ menos niveles.
  \item \hl{SIMD}: con una instrucci\'on AVX prueba \hl{varias cajas a la vez}.
  \item \hl{Layout} compacto y nodos cuantizados $\Rightarrow$ menos fallos de cach\'e.
  \item Calidad de build seleccionable: \texttt{HIGH} (SBVH) / \texttt{MEDIUM} / \texttt{LOW},
        que mapeamos a SAH / Median / Morton.
\end{itemize}
\end{column}
\begin{column}{0.46\textwidth}
\begin{block}{Por qu\'e es tan r\'apido}
\begin{itemize}\itemsep2pt
  \item \alert{No} por un \'arbol mejor: su calidad es comparable a nuestra SAH.
  \item S\'\i{} por la \hl{ingenier\'ia del kernel} (SIMD $+$ MBVH $+$ cach\'e).
\end{itemize}
\end{block}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Setup experimental}
\begin{columns}[T]

\textbf{Plataforma}
\begin{itemize}\itemsep1pt
  \item CPU: \hl{AMD Ryzen 7 5700X3D} (8 n\'ucleos, 16 hilos).

\textbf{Escena:} campo de esferas (densidad variable):
\begin{itemize}\itemsep1pt
  \item Small \hl{12\,812} \textbullet{} Medium \hl{51\,212} \textbullet{} Large \hl{204\,812} tri.
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\textbf{Metodolog\'ia} (reproducible: \texttt{-{}-bench})
\begin{itemize}\itemsep1pt
  \item 3 frames de \emph{warmup} $+$ promedio de 10.
  \item \hl{render/frame}: frame completo (CPU y GPU).
  \item \hl{traversal}: rayos primarios, 1 hilo (CPU y Embree).
  \item \alert{nodos/rayo}: calidad del \'arbol, \emph{independiente del hardware}.
\end{itemize}
\end{column}
\end{columns}
\vspace{2pt}
\centering\footnotesize\textcolor{mute}{Matriz completa: 3 backends $\times$ 3 heur\'isticas $\times$ 3 tama\~nos.}

\begin{frame}{Resultado principal \textemdash{} escena Large (204\,812 tri)}
\begin{table}\centering\small
\setlength{\tabcolsep}{8pt}
\renewcommand{\arraystretch}{1.25}
\begin{tabular}{llrrr}
\toprule
\textbf{Backend} & \textbf{M\'etrica} & \textbf{SAH} & \textbf{Median} & \textbf{Morton} \\

\'Arbol           & nodos/rayo        & \hl{138} & 265 & 660 \\
\bottomrule
\end{tabular}

\vspace{2pt}
\begin{itemize}\itemsep2pt
  \item \hl{Heur\'istica:} SAH visita $\sim$4{,}8$\times$ menos nodos que Morton $\Rightarrow$ CPU $\sim$5{,}8$\times$ m\'as r\'apido.
  \item \hl{GPU:} mismo \'arbol, $\sim$5{,}7$\times$ sobre CPU con SAH (SIMT, un p\'ixel por hilo).
  \item \hl{Embree:} frame casi constante ante la heur\'istica \textemdash{} usa su \emph{propio} \'arbol robusto.
\end{itemize}
\end{frame}

\begin{frame}{Conclusiones}
\begin{itemize}\itemsep4pt
  \item Ray tracer con BVH en \hl{tres backends} (CPU, GPU OpenCL, Embree) y
        \hl{benchmark reproducible} sobre la misma geometr\'ia.
  \item \hl{Heur\'istica:} la \alert{SAH} da el mejor \'arbol
        ($\sim$4{,}8$\times$ menos nodos/rayo que Morton) $\Rightarrow$ render $\sim$5{,}8$\times$ m\'as r\'apido.
  \item \hl{GPU:} el \emph{flatten} $+$ pila expl\'icita $+$ SIMT logran \hl{4\,--\,10$\times$} sobre CPU
        sin hardware RT dedicado.
  \item \hl{Embree:} su ventaja est\'a en el \emph{recorrido SIMD}, no en el \'arbol.
  \item \hl{Din\'amico:} probamos que rebuild-por-frame \emph{no} vuelca la balanza a Morton;
        el traversal domina a resoluci\'on de render.
\end{itemize}
\end{frame}
